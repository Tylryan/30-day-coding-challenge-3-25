

fun fib(n) {
    if (n <= 1) {
        return n;
    }

    return fib(n - 1) + fib(n - 2);
}

fun makeCounter() {
    var i = 0;
    fun count() {
        i = i + 1;
        return i;
    }

    return count;
}
